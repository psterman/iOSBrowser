### 解决方案
```swift
// 添加了 HotTrendItem 扩展
extension HotTrendItem {
    var displayRank: String {
        switch rank {
        case 1: return "🥇"
        case 2: return "🥈" 
        case 3: return "🥉"
        default: return "\(rank)"
        }
    }
    
    var isTopThree: Bool {
        return rank <= 3
    }
}
```
